二維數組回形遍歷 
n, m = map(int,input().split())
a = []
b = []

for j in range(0,n):
    b = list(map(int,input().split()))
    a.append(b)
r = 0
c = 1
cnt =0

while(True):
    for i in range(0+r,m-r):
        print(a[r][i])
        cnt+=1
    if(cnt == n*m):
        break
    for i in range(0+c,n-c):
        print(a[i][m-1-r])
        cnt+=1
    if(cnt == n*m):
        break
    for i in range(0+r,m-r):
        print(a[n-1-r][m-1-i])
        cnt+=1
    if(cnt == n*m):
        break
    for i in range(0+c,n-c):
        print(a[n-1-i][r])
        cnt+=1
    if(cnt == n*m):
        break
    r += 1
    c += 1
